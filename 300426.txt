    public class Student
    {
        public string Name { get; set; } // Автоматическое свойство для Name, get — чтение и set — запись.

        private int  _age; // приватное поле 
        
        public int Age
        {
            get =>_age; //get — возвращает значение 
            
            set // set - проверка 
            
            {
                if(value >= 0)
                {
                    _age = value; // value — ключевое слово, содержащее передаваемое значение.
                }
                else 
                {
                    throw new ArgumentException("Возраст не может быть отрицательным!");
                }
            }
            
        }

        public void Print()
        {
            Console.WriteLine(Name);
            Console.WriteLine(_age);
        }

        
        public Student(string name = "name", int age = 0) //конструктор
        {
            Name = name; // устанавливем значение для Name 
            Age = age; // устанавливем значение для Age  проходит валидацию через set

        }
        public Student()
        {
            Name = "Ivan Zolo"; // устанавливем значение для Name 
            Age = 5;
        }
        
        ~Student()
        {
            Console.WriteLine("эх блин");
        }
    }
